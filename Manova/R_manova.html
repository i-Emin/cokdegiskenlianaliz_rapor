# Levene's Test sonuçlarını saklamak için boş bir liste
levene_results <- list()

# Her bağımlı değişken için Levene's Test
for (var in dependent_vars) {
  if (var %in% colnames(data)) {
    test_result <- leveneTest(as.formula(paste(var, "~", independent_var)), data = data)
    levene_results[[var]] <- test_result
  }
}

# Test sonuçlarını bir araya getirme
levene_summary <- lapply(names(levene_results), function(var) {
  data.frame(
    Variable = var,
    F_Value = levene_results[[var]]$`F value`[1],
    P_Value = levene_results[[var]]$`Pr(>F)`[1]
  )
}) %>%
  bind_rows()

# Sonuçları yazdırma
print("Levene's Test Sonuçları:")
